\begin{question} {\textbf{Prove that given $R$ and $S$ are transitive, $R \cap S$ is transitive.}}
    \begin{part} Unrolling, we get... \\ \\
    $(a,b) \in R \cap S \wedge (b,c) \in R \cap S \rightarrow (a,c) \in R \cap S$
    \end{part}

    \begin{part}
    
    \begin{minipage}{40em}
    \begin{proof}\leavevmode
    \begin{enumerate}
        \item $(a,b) \in R \wedge (b,c) \in R\rightarrow (a,c) \in R$ \hfill (Given)
        \item $(a,b) \in S \wedge (b,c) \in S\rightarrow (a,c) \in S$ \hfill (Given)

        
        \begin{level}
        \begin{enumerate}[label=3.\arabic*.]
            % \item 
        
            \item $(a,b) \in R \cap S \wedge (b,c) \in R \cap S$ \hfill (Assumption)
            \item $(a,b) \in R \cap S$ \hfill (Elim $\wedge$: 1.1)
            \item $(b,c) \in R \cap S$ \hfill (Elim $\wedge$: 1.1)
            \item $(a,b) \in R \wedge (a,b) \in S$ \hfill (Def of Union: 1.2)
            \item $(a,b) \in R$ \hfill (Elim $\wedge$ 1.4)
            \item $(a,b) \in S$ \hfill (Elim $\wedge$ 1.4)
            \item $(b,c) \in R \wedge (b,c) \in S$ \hfill (Def of Union: 1.3)
            \item $(b,c) \in R$ \hfill (Elim $\wedge$ 1.7)
            \item $(b,c) \in S$ \hfill (Elim $\wedge$ 1.7)
            \item $(a,b) \in R \wedge (b,c) \in R$ \hfill (Intro $\wedge$: $1.5, 1.8$)
            \item $(a,b) \in S \wedge (b,c) \in S$ \hfill (Intro $\wedge$: $1.6, 1.9$)
            \item $(a,c) \in R$ \hfill (Modus Ponens: $1, 1.10$)
            \item $(a,c) \in S$ \hfill (Modus Ponens: $2, 1.11$)
            \item $(a,c) \in R \cap S$ \hfill (Def of Union)
        \end{enumerate}
        \end{level}

    \item     $(a,b) \in R \cap S \wedge (b,c) \in R \cap S \rightarrow (a,c) \in R \cap S$ \hfill (Direct Proof)

        
    \end{enumerate}
    \end{proof}
